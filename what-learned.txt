it on multiple element or input field we need to use this hook multiple time;
    ===> 
        const nameRef = useRef();
        <form>
            <input ref={nameRef} />
        </form>

2. we need to add this hook var on field ref props.

3. we need to use value property on current property on hook var to get this field data.
    ===>
        console.log(nameRef.current.value);

4. we need to send data as key value pair on dispatch() method  object arg. to store 
it on redux store. this obj must have type property.
    ===>
        dispatch({type:"ADD_PATIENT", name:"Md. Test", id:1});

5. we need to access this data on action obj arg of reducer method. 
    ===>
        export const patientState = {
            patients:[]
        };

        export const patientReducer = (state, action) => {
            switch(action.type){
                case "ADD_PATIENT":
                    const newPatient = {
                        id: action.id,
                        name: action.name,
                    }
                    const allPatients = [...state.patients, newPatient];
                    return {patients:allPatients}; 
                case "REMOVE_PATIENT":
                    const remaining = state.patients.filter(pt => pt.id !== action.id);
                    const newState = {patients:remaining};
                    return newState;
                default:
                    return state;
            }
        }
